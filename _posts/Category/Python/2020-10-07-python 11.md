# 11일차. PK, FK 관련 sQL

```sql
USE hanbitDB;
DROP TABLE student;
CREATE TABLE student (

  s_id  CHAR(10) ,
  s_name CHAR(5),
  s_addr VARCHAR(100) ,
  s_phone CHAR(11) ,
  s_dept VARCHAR(10) ,
  s_eYear SMALLINT ,
  s_subj  CHAR(10) , 
  s_sea   CHAR(1) , -- P, S, F, W
  s_score CHAR(2), 
  s_sukcha SMALLINT

);
```

DROP TABLE student;
CREATE TABLE student (

  s_id  CHAR(10) PRIMARY KEY ,
  s_name CHAR(5),
  s_addr VARCHAR(100) ,
  s_phone CHAR(11) ,
  s_dept VARCHAR(10) ,
  s_eYear SMALLINT );

DROP TABLE subj;
CREATE TABLE subj(

  s_num  INT  auto_increment primary KEY,

  st_id CHAR(10) ,
  s_subj  CHAR(10) , 
  s_sea   CHAR(1) , -- P, S, F, W
  s_score CHAR(2), 
  s_sukcha SMALLINT ,
  FOREIGN KEY(st_id) REFERENCES student(s_id));
